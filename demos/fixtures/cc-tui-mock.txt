  ❯ Write a small JavaScript debounce function with a maxWait option.

● Here's a debounce that fires the last call after `wait` ms of
  quiet, with a `maxWait` upper bound so it can't be deferred
  forever under continuous traffic.

  function debounce(fn, wait, { maxWait } = {}) {
    let timer = null;
    let maxTimer = null;
    let pending = null;

    const invoke = () => {
      if (timer) clearTimeout(timer);
      if (maxTimer) clearTimeout(maxTimer);
      timer = maxTimer = null;
      const args = pending; pending = null;
      if (args) fn(...args);
    };

    return function debounced(...args) {
      pending = args;
      if (timer) clearTimeout(timer);
      timer = setTimeout(invoke, wait);
      if (maxWait != null && maxTimer == null) {
        maxTimer = setTimeout(invoke, maxWait);
      }
    };
  }

  `wait` resets on every call; `maxWait` guarantees the function
  fires at least once per that interval even under continuous calls.
